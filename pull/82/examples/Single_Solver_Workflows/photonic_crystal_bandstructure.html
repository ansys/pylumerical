Photonic Crystal Bandstructure (FDTD)#

This example demonstrates a photonic crystal simulation utilizing Structure and Analysis Group objects. Based on: https://optics.ansys.com/hc/en-us/articles/360041566614-Rectangular-Photonic-Crystal-Bandstructure

In Part 1, we build the structure and set the FDTD simulation region. In this case, the spheres are holes (filled with air, n = 1) and the background material is a simple dielectric material. Some advanced simulation objects, including the dipole cloud source and bandstructure analysis groups, are imported from the Object Library. We run a single simulation and visualize the resulting spectrum.

In Part 2, we set up a series of sweeps to collect the resonant frequencies. In this example, we use the built-in sweep tool in Lumerical, but the parameter sweeps could also be set up from Python. We then run the sweeps and plot the results.

Prerequisites: Valid FDTD license is required.

Part 1: Set up structures and simulation objects

 1# Define parameters
 2
 3# Set filename for saving and loading
 4filename = "photonic_crystal_bandstructure.fsp"
 5
 6# set desired period of the array (3D)
 7ax = 0.5e-6
 8ay = 0.5e-6
 9az = 0.5e-6
10
11# number of spheres in each direction
12nx = 3
13ny = 3
14nz = 3
15
16sphere_radius = 0.25e-6
17sphere_index = 1  # Specify index if "Object defined dielectric" is used; otherwise, specify material from database below
18sphere_material = "etch"  # Etch n = 1 (air)
19background_index = 3.6055
20
21# Define frequencies / wavelengths of interest
22f1 = 1e12  # THz
23f2 = 220e12  # THz

1[ ]:

 1# Initialize session and build simulation objects. Set hide = True to hide the Lumerical GUI.
 2# This block will build an array of spheres for the photonic crystal structure.
 3# We add an FDTD region, sources, and monitors. The sources and monitors are imported from the Object Library.
 4# Finally, the simulation file is saved to the file name set earlier.
 5
 6with lumapi.FDTD(hide=False) as fdtd:
 7    # Add rectangular lattice array of spheres
 8    # Note you can also use Lumerical's built-in "rect_pc_3D" object using fdtd.addobject("rect_pc_3D")
 9    # In this example, we construct the array programmatically
10    fdtd.addstructuregroup({"name": "rect_pc_3D"})
11    for ix, iy, iz in itertools.product(*map(range, (nx, ny, nz))):
12        objname = f"c{ix}{iy}{iz}"
13        fdtd.addsphere(
14            {"name": objname, "x": (ix - 1) * ax, "y": (iy - 1) * ay, "z": (iz - 1) * az, "radius": sphere_radius, "material": sphere_material}
15        )
16        fdtd.addtogroup("rect_pc_3D")  # Adds the recently created sphere to the structure group
17
18    # Add FDTD region
19    dx = 0.025e-6  # Mesh dx, dy, dz
20    fdtd_geometry_props = {"x": 0, "x span": ax, "y": 0, "y span": ay, "z": 0, "z span": az}
21    fdtd_mesh_props = {"index": background_index, "mesh type": "uniform", "dx": dx, "dy": dx, "dz": dx}
22    fdtd_boundary_props = {"x min bc": "Bloch", "y min bc": "Bloch", "z min bc": "Bloch", "set based on source angle": False, "bloch units": "SI"}
23    # Combine properties settings into one dictionary
24    fdtd_props = OrderedDict({**fdtd_geometry_props, **fdtd_mesh_props, **fdtd_boundary_props})
25
26    fdtd.addfdtd(properties=fdtd_props)
27
28    # Set up sources (dipole cloud) and monitors (bandstructure)
29    # These are both Analysis Groups available from the object library
30    dipole_geometry_props = {"n dipoles": 3, "lattice type": 3, "z span": az, "ax": ax, "ay": ay, "az": az, "a": az, "x": 0, "y": 0, "z": 0}
31    dipole_frequency_props = {"f1": f1, "f2": f2, "kx": 0.5, "ky": 0, "kz": 0}
32    dipole_props = OrderedDict({**dipole_geometry_props, **dipole_frequency_props})
33
34    fdtd.addobject("dipole_cloud", properties=dipole_props)
35
36    bandstructure_geometry_props = {"n monitors": 10, "x": 0, "x span": ax, "y": 0, "y span": ay, "z": 0, "z span": az}
37    bandstructure_frequency_props = {"f1": f1, "f2": f2}
38    bandstructure_props = OrderedDict({**bandstructure_geometry_props, **bandstructure_frequency_props})
39    fdtd.addobject("bandstructure", properties=bandstructure_props)
40
41    # zoom CAD view around simulation region
42    fdtd.select("FDTD")
43    fdtd.setview("extent")
44
45    fdtd.save(filename)
46    print("File saved to folder as: " + filename)

 1# Open the file and run a single simulation. Visualize the spectrum.
 2
 3with lumapi.FDTD(filename, hide=False) as fdtd:
 4    print("Simulation running...")
 5    fdtd.run()
 6    print("Run completed. Analyzing and plotting results...")
 7    fdtd.runanalysis()
 8
 9    # Plot results
10    single_spectrum = fdtd.getresult("bandstructure", "spectrum")
11    spectrum = single_spectrum["fd"]
12    frequencies = single_spectrum["f"]
13
14    plt.figure()
15    plt.plot(frequencies, spectrum)
16    plt.title("Single Simulation Spectrum")
17    plt.xlabel("Frequency")
18    plt.show(block=False)
19    plt.pause(2)

Part 2: Set up and run sweeps to extract resonant frequencies and plot the bandstructure

1[ ]:

 1# Normalization factor for SI units; see note above.
 2norm_x = 2 * np.pi / ax
 3norm_y = 2 * np.pi / ay
 4norm_z = 2 * np.pi / az
 5
 6# The total number of simulations will be num_points*3
 7num_points = 5
 8
 9
10# Define a function to add the sweeps
11def add_bandstructure_sweep(sweep_name, x_start, x_stop, y_start, y_stop, z_start, z_stop):
12    """Set up a sweep for bandstructure calculations."""
13    fdtd.addsweep()
14    fdtd.setsweep("sweep", "name", sweep_name)
15    fdtd.setsweep(sweep_name, "type", "Ranges")
16    fdtd.setsweep(sweep_name, "number of points", num_points)
17
18    # set the sweep properties
19    props_kx = {"Name": "kx", "Parameter": "::model::FDTD::kx", "Type": "Number", "Start": x_start, "Stop": x_stop}
20    fdtd.addsweepparameter(sweep_name, props_kx)
21    props_ky = {"Name": "ky", "Parameter": "::model::FDTD::ky", "Type": "Number", "Start": y_start, "Stop": y_stop}
22    fdtd.addsweepparameter(sweep_name, props_ky)
23    props_kz = {"Name": "kz", "Parameter": "::model::FDTD::kz", "Type": "Number", "Start": z_start, "Stop": z_stop}
24    fdtd.addsweepparameter(sweep_name, props_kz)
25
26    # define results
27    result_fs = {"Name": "fs", "Result": "::model::bandstructure::fs"}
28    fdtd.addsweepresult(sweep_name, result_fs)
29    result_spectrum = {"Name": "fs", "Result": "::model::bandstructure::spectrum"}
30    fdtd.addsweepresult(sweep_name, result_spectrum)
31
32
33# Now add the sweeps to the file
34with lumapi.FDTD(filename, hide=False) as fdtd:
35    # Add Gamma-X sweep
36    add_bandstructure_sweep("Gamma-X", 0 * norm_x, 0.5 * norm_x, 0 * norm_y, 0 * norm_y, 0 * norm_z, 0 * norm_z)
37    # Add X-M sweep
38    add_bandstructure_sweep("X-M", 0.5 * norm_x, 0.5 * norm_x, 0 * norm_y, 0.5 * norm_y, 0 * norm_z, 0 * norm_z)
39    # Add M-R sweep
40    add_bandstructure_sweep("M-R", 0.5 * norm_x, 0.5 * norm_x, 0.5 * norm_y, 0.5 * norm_y, 0 * norm_z, 0.5 * norm_z)
41    fdtd.save(filename)
42    print("Sweeps have been set up.")

1[ ]:

1# Now run all the sweeps - this may take a few minutes
2with lumapi.FDTD(filename, hide=False) as fdtd:
3    print("Running sweeps...")
4    fdtd.runsweep()
5    fdtd.save(filename)
6    # If you have previously run the sweeps, you can load them using this line instead:
7    # fdtd.loadsweep()
8print("Sweeps have been run.")

 1# Retrieve and analyze data from the sweeps
 2with lumapi.FDTD(filename, hide=False) as fdtd:
 3    fs_all = np.zeros((50, 3 * num_points))  # 50 is the number of frequencies, and there are 3*num_points bandstructure points
 4
 5    sweepname = "Gamma-X"
 6    resonance = fdtd.getsweepresult(sweepname, "fs")  # Results are returned as Python dict
 7    resonance_fs = resonance["fs"]
 8    fs_all[0:50, 0:num_points] = resonance_fs
 9
10    sweepname = "X-M"
11    resonance = fdtd.getsweepresult(sweepname, "fs")  # Results are returned as Python dict
12    resonance_fs = resonance["fs"]
13    fs_all[0:50, num_points : 2 * num_points] = resonance_fs
14
15    sweepname = "M-R"
16    resonance = fdtd.getsweepresult(sweepname, "fs")  # Results are returned as Python dict
17    resonance_fs = resonance["fs"]
18    fs_all[0:50, 2 * num_points : 3 * num_points] = resonance_fs

1k = np.linspace(1, 3 * num_points, 3 * num_points)
2c = 3 * 10**8
3plt.figure()
4for f in range(0, 50):
5    plt.scatter(k, fs_all[f] * ax / c)
